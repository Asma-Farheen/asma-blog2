## ðŸ”¹ useState â€” Manage Local State

`useState` lets a component remember and update values between renders.  
Itâ€™s used to manage local data like counters, forms, or toggles.

**Example:**
```js
import { useState, useEffect, useContext, createContext, useRef, useMemo, useCallback } from "react";

function HooksExample() {
  // useState â€” Manage Local State
  const [count, setCount] = useState(0);
  const [number, setNumber] = useState(5);

  // Update the value
  setCount(count + 1);

  // ðŸ”¹ useEffect â€” Handle Side Effects
  useEffect(() => {
    console.log("Component mounted or updated");
  }, []);

  // ðŸ”¹ useContext â€” Share Data Without Props

  const theme = useContext(ThemeContext);

  // ðŸ”¹ useRef â€” Access DOM or Store Persistent Values

  const renderCount = useRef(0);
  renderCount.current += 1;

  function focusInput() {
    inputRef.current.focus();
  }

  // ðŸ”¹ useMemo â€” Optimize Expensive Calculations
  const factorial = useMemo(() => {
    let result = 1;
    for (let i = 1; i <= number; i++) {
      result *= i;
    }
    return result;
  }, [number]);

  // ðŸ”¹ useCallback â€” Cache Functions for Performance
  const increment = useCallback(() => {
    setCount(c => c + 1);
  }, []);

  return (
    <div>
      <h2>React Hooks Example</h2>
      <p>Current Count: {count}</p>
      <p>Factorial of {number}: {factorial}</p>
      <p>Theme from Context: {theme}</p>
      <input ref={inputRef} placeholder="Focus me using Ref" />
      <p>Render Count: {renderCount.current}</p>

      <button onClick={increment}>Increment Count</button>
      <button onClick={() => setNumber(n => n + 1)}>Increase Number</button>
      <button onClick={focusInput}>Focus Input</button>
    </div>
  );
}
